console=ttyO0,115200n8
serverip=192.167.1.1
loadaddr=0x82000000
fdtaddr=0x88000000
rootpath=/srv/nfs/rootfs,nolock,wsize=1024,rsize=1024 rootwait rootdelay=5
loadfromsd=load mmc 0:2 ${loadaddr} /boot/uImage;load mmc 0:2 ${fdtaddr} /boot/am335x-boneblack.dtb
nfsbootargs=setenv bootargs console=${console} root=/dev/nfs rw nfsroot=${serverip}:${rootpath}
linuxbootargs=setenv bootargs console=${console} root=/dev/mmcblk0p2 rw 
uenvcmd=setenv autoload no; run loadfromsd; if ${netbootcmd}; then echo booting from nfs; run nfsbootargs; else echo booting from sd; run linuxbootargs; fi; bootm ${loadaddr} - ${fdtaddr}
